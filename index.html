const counter = document.querySelector('#counter');
const plusBtn = document.querySelector('#plus-btn');
const minusBtn = document.querySelector('#minus-btn');
const setCounter = () => counter.innerText = count;
const raiseCount = () => count++;
const lowerCount = () => count--;
plusBtn.onclick = () => {
    raiseCount();
    setCounter();
};
minusBtn.onclick = () => {
    lowerCount();
    setCounter();
}

<script>
    const todos = ['learn', 'chat with gpt'];
    const todoList = document.querySelector('#todo-list');

    const renderTodos = () => {
        todoList.innerHTML = '';
        todos.forEach(todo => {
            const p = document.createElement('p');
            p.innerHTML = todo;
            p.classList.add('todo');

            const button = document.createElement('button');
            button.innerText = 'Delete Todo';
            button.classList.add('todoBtn');

            todoList.appendChild(p);
            todoList.appendChild(button);
        });
        const deleteBtns = document.querySelectorAll('.todoBtn');

        deleteBtns.forEach((deleteBtn, index) => {
            deleteBtn.onclick = () => {
                todos.splice(index, 1);
                renderTodos();
            }
        });
    };
    renderTodos();

    const addTodo = document.querySelector('#add-todo');
    const addTodoBtn = document.querySelector('#add-todo-btn');

    addTodoBtn.onclick = e => {
        e.preventDefault();
        todos.push(addTodo.value);
        renderTodos();
    };
</script>

<script type="text/babel">
    const ReactTodoList = () => {
        const [todoList, setTodoList] = React.useState(['Learn', 'Chat with GPT']);
        const [newTodo, setNewTodo] = React.useState('');

        const addTodo = () => setTodoList( [...todoList, newTodo] );
        const removeTodo = index => setTodoList(todoList.filter((todo, i) => i !== index));

        return (
            <div>
                <label htmlFor="newTodo">New Todo</label>
                <input type="text" id="newTodo" name="newTodo" onChange={ e => setNewTodo(e.target.value) } />
                <button onClick={ addTodo }>Add Todo</button>
                {
                    todoList.map((todo, index) => (
                        <div key={ index }>
                            <p>{ todo }</p>
                            <button onClick={ () => removeTodo(index) }>Delete</button>
                        </div>
                    ))
                }
            </div>
        )
    }
    ReactDOM.createRoot(document.querySelector('#react-todo')).render(<ReactTodoList />);
</script>
